fn main() -> Result<(), ffmpeg::Error> {
    ffmpeg::init()?;

    // Usage: cargo run -- input.mp4 5.0 out.ppm
    let args: Vec<String> = env::args().collect();
    let _input_path = &args[1];
    let target_seconds: f64 = args[2].parse().unwrap();
    let output_path = &args[3];

    let mut ictx = ffmpeg::format::input(&"assets/13232364_3840_2160_24fps.mp4")?;

    let input_stream = ictx
        .streams()
        .best(ffmpeg::media::Type::Video)
        .ok_or(ffmpeg::Error::StreamNotFound)?;
    let video_stream_index = input_stream.index();
    let time_base = input_stream.time_base(); // <-- save this, we need it

    let context_decoder =
        ffmpeg::codec::context::Context::from_parameters(input_stream.parameters())?;
    let mut decoder = context_decoder.decoder().video()?;

    // === THE SEEK ===
    // Convert target seconds → stream time_base units
    // pts = seconds / time_base = seconds × (time_base.denominator / time_base.numerator)
    let target_pts =
        (target_seconds * time_base.denominator() as f64 / time_base.numerator() as f64) as i64;

    // Seek the container. This positions the demuxer at the nearest
    // keyframe ≤ target_pts in the video stream.
    ictx.seek(target_pts, ..target_pts)?;

    // IMPORTANT: flush the decoder's internal buffers after seeking.
    // Otherwise it'll spit out stale frames from before the seek.
    decoder.flush();

    // === DECODE FORWARD until we land on/past target ===
    let mut decoded = ffmpeg::frame::Video::empty();
    let mut found: Option<ffmpeg::frame::Video> = None;

    'outer: for (stream, packet) in ictx.packets() {
        if stream.index() != video_stream_index {
            continue;
        }
        decoder.send_packet(&packet)?;

        while decoder.receive_frame(&mut decoded).is_ok() {
            let frame_pts = decoded.pts().unwrap_or(0);
            if frame_pts >= target_pts {
                // This is our frame (or the first one after target)
                found = Some(decoded.clone());
                break 'outer;
            }
            // else: keep decoding forward, we haven't reached target yet
        }
    }

    let frame = found.ok_or(ffmpeg::Error::Eof)?;

    // === Convert YUV → RGB so we can save it ===
    let mut scaler = Scaler::get(
        frame.format(),
        frame.width(),
        frame.height(),
        Pixel::RGB24,
        frame.width(),
        frame.height(),
        Flags::BILINEAR,
    )?;

    let mut rgb_frame = ffmpeg::frame::Video::empty();
    scaler.run(&frame, &mut rgb_frame)?;

    // Save as PPM (dead simple format — no PNG crate needed)
    save_ppm(&rgb_frame, output_path).unwrap();
    println!(
        "saved frame at pts={} to {}",
        frame.pts().unwrap_or(0),
        output_path
    );

    Ok(())
}

fn save_ppm(frame: &ffmpeg::frame::Video, path: &str) -> std::io::Result<()> {
    let mut file = File::create(path)?;
    writeln!(file, "P6\n{} {}\n255", frame.width(), frame.height())?;
    // data(0) is the RGB plane; stride may be > width*3 due to alignment
    let data = frame.data(0);
    let stride = frame.stride(0);
    let row_bytes = (frame.width() * 3) as usize;
    for y in 0..frame.height() as usize {
        let start = y * stride;
        file.write_all(&data[start..start + row_bytes])?;
    }
    Ok(())
}
